```jsx
function SideNav(props) {
  if (props.isMobile) {
    return <div className="side-styling">Navigation</div>;
  } else {
    return <div className="bottom-styling">Navigation</div>;
  }

```

```jsx
function LeftNav(props) {
  return (
    <div>
      Navigation
      <ul>
        <li>First Link</li>
        <li>Second Link</li>
      </ul>
    </div>
  );
}

function BottomNav(props) {
  return (
    <div style={{ display: 'flex' }}>
      <div>First Link</div>
      <div>Second Link</div>
    </div>
  );
}

function SideNav(props) {
  if (props.isMobile) {
    return <LeftNav {...props} />;
  } else {
    return <BottomNav {...props} />;
  }
}
```

```jsx
function SideNav(props) {
  let theme = useTheme();
  let isSmall = useMediaQuery(theme.breakpoints.down('sm'));

  if (isSmall) {
    return <div className="side-styling">Navigation</div>;
  } else {
    return <div className="bottom-styling">Navigation</div>;
  }
}
```

# A pattern I use a lot

<CodeSurfer>

```jsx
{
  /* to conditionally render stuff*/
}
function SomeComponent() {
  let theme = useTheme();
  let isSmaller = useMediaQuery(theme.breakpoints.down('md'));

  if (isSmaller) {
    return <div>Some stuff</div>;
  } else {
    return <div>Some other stuff</div>;
  }
}
```
