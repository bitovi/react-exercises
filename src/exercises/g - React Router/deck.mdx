# Packages react-router, and his cousins

React Router has been split into a couple of packages so that the guts
are in react-router, and differences in environment have been moved to
react-router-dom, and react-router-native. Basically for us, we will always be using "react-router-dom"

```jsx
import { BrowserRouter, Switch, Route } from 'react-router-dom';

export default function App() {
  return (
    <BrowserRouter>
      {/* A <Switch> looks through its children <Route>s and
            renders the first one that matches the current URL. */}
      <Switch>
        <Route path="/about" component={About} />
        <Route path="/workout/:id" component={Workout} />
        <Route exact path="/" component={Home} />
        <Route path="/" component={NoMatch} />
      </Switch>
    </BrowserRouter>
  );
}
```

```jsx
import { BrowserRouter, Switch, Route, useParams } from 'react-router-dom';

export default function App() {
  return (
    <BrowserRouter>
      {/* A <Switch> looks through its children <Route>s and
            renders the first one that matches the current URL. */}
      <Switch>
        <Route path="/about" component={About} />
        <Route path="/workout/:id" component={Workout} />
        <Route exact path="/" component={Home} />
        <Route path="/" component={NoMatch} />
      </Switch>
    </BrowserRouter>
  );
}

function Workout() {
  let { id } = useParams();
  return <div>{id}</div>;
}
```

```jsx
import React from 'react';
```
